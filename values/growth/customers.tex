% customers.tex - Customer metrics, churn rates, and subscriber projections

%% ================= Churn Rates =================
% Annual churn rate by year (%)
\newcommand{\churnYearOne}{5}
\newcommand{\churnYearTwo}{4.5}
\newcommand{\churnYearThree}{4}

% Average annual churn rate across 3 years
\newcommand{\avgAnnualChurn}{\fpeval{(\churnYearOne + \churnYearTwo + \churnYearThree) / 3 / 100}}

%% ================= Customer Acquisition (Direct Paid Only) =================
% New customers from paid acquisition (marketing budget / CAC)
\newcommand{\baseNewSubsYearOne}{\fpeval{round(\marketingBudgetYearOne / \cacDigital, 0)}}
\newcommand{\baseNewSubsYearTwo}{\fpeval{round(\marketingBudgetYearTwo / \cacDigital, 0)}}
\newcommand{\baseNewSubsYearThree}{\fpeval{round(\marketingBudgetYearThree / \cacDigital, 0)}}

% Hardware customers from paid acquisition
\newcommand{\baseHwCustomersYearTwo}{800}
\newcommand{\baseHwCustomersYearThree}{1500}

%% ================= Total New Customers (Direct Acquisition Only) =================
% New subscribers in given year
\newcommand{\newSubsYearOne}{\baseNewSubsYearOne}
\newcommand{\newSubsYearTwo}{\baseNewSubsYearTwo}
\newcommand{\newSubsYearThree}{\baseNewSubsYearThree}

% Hardware customers in given year
\newcommand{\hwCustomersYearTwo}{\baseHwCustomersYearTwo}
\newcommand{\hwCustomersYearThree}{\baseHwCustomersYearThree}

%% ================= Cumulative Active Subscribers =================
% Total active subscribers at end of year (after churn)
% Year 1: New subscribers only
\newcommand{\totalSubsYearOne}{\fpeval{round(\newSubsYearOne, 0)}}

% Year 2: Year 1 × retention + new subscribers
\newcommand{\totalSubsYearTwo}{\fpeval{round(\totalSubsYearOne * (1 - \churnYearOne/100) + \newSubsYearTwo, 0)}}

% Year 3: Year 2 × retention + new subscribers
\newcommand{\totalSubsYearThree}{\fpeval{round(\totalSubsYearTwo * (1 - \churnYearTwo/100) + \newSubsYearThree, 0)}}

%% ================= Customer Acquisition Efficiency =================
% Customers acquired per $1000 marketing spend
\newcommand{\custPerThousandYearOne}{50}
\newcommand{\custPerThousandYearTwo}{50}
\newcommand{\custPerThousandYearThree}{50}

%% ================= Beta Launch Metrics =================
% Beta launch target users
\newcommand{\betaUsers}{1000}
