% subscriptions.tex - Subscription product definitions and tier structure
% This file consolidates all subscription-related values

%% ================= Subscription Tiers Overview =================
% COMPUTATION MARKETPLACE TIERS:
% Tiers represent different computation job durations (hours):
% Basic = 2 hours, Medium = 24 hours, Professional = 48 hours,
% Golden = 168 hours, Platinum = 336 hours
% All jobs recur weekly (4.28 times per month)

%% ================= Subscription Product Mix =================
% Note: Mix percentages defined in values/economics/margins.tex
% Basic:        35% - Entry level users
% Medium:       40% - Core user base
% Professional: 15% - Power users
% Golden:        9% - Enterprise users
% Platinum:      1% - Premium enterprise

%% ================= Market Size - Subscriptions =================
% TAM Components (estimated user counts)
\newcommand{\tamBitcoinUsers}{3000000}      % Bitcoin security-conscious users
\newcommand{\tamPasswordMgrUsers}{4000000}   % Password manager users seeking stronger security
\newcommand{\tamPrivateSecUsers}{1500000}    % Private security/insurance customers
\newcommand{\tamPhysicalVaultUsers}{500000}  % Physical vault users transitioning to digital

% Total Addressable Market (count) - sum of all segments
\newcommand{\tamSubsGlobal}{\fpeval{\tamBitcoinUsers + \tamPasswordMgrUsers + \tamPrivateSecUsers + \tamPhysicalVaultUsers}}

% SAM as percentage of TAM
\newcommand{\samPercentOfTam}{15}  % 15% of TAM is serviceable

% Serviceable Available Market (calculated from TAM)
\newcommand{\samSubs}{\fpeval{round(\tamSubsGlobal * \samPercentOfTam / 100, -5)}}

% Target penetration of SAM (%)
\newcommand{\targetShareSubs}{15}  % 3.5% of SAM

% Target customer count for subscriptions (thousands)
\newcommand{\targetSubsCountk}{\fpeval{round(\samSubs * \targetShareSubs / 100 / 1000, 0)}}

%% ================= Operational Metrics =================
% CAC reduction percentage from merchandise
% Formula: (merchAttachRate * merchAvgPrice * merchAvgMargin / 100) / cacDigital * 100
% Using values from merchandise.tex and acquisition.tex
\newcommand{\cacReductionPercent}{\fpeval{round(\merchAttachRate * \merchAvgPrice * \merchAvgMargin / 100 / \cacDigital * 100, 1)}}

% Combined TAM in thousands (subscriptions + hardware)
\newcommand{\combinedTAMk}{\fpeval{round((\tamSubsGlobal + \tamHwGlobal) / 1000, 0)}}
