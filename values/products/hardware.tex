%% ================= Market Size - Hardware =================
% TAM Components for hardware (estimated user counts)
\newcommand{\tamHwBitcoinPower}{1500000}     % Bitcoin power users wanting self-sovereign hardware
\newcommand{\tamHwPasswordMgrUpgrade}{800000} % Password mgr users seeking hardware upgrade
\newcommand{\tamHwPrivateSec}{600000}        % Private security customers wanting physical device
\newcommand{\tamHwPhysicalVault}{600000}     % Physical vault users (direct replacement market)

% Total Addressable Market (count) - sum of all segments
\newcommand{\tamHwGlobal}{\fpeval{\tamHwBitcoinPower + \tamHwPasswordMgrUpgrade + \tamHwPrivateSec + \tamHwPhysicalVault}}

% SAM as percentage of hardware TAM
\newcommand{\samHwPercentOfTam}{10}  % 10% - hardware has lower SAM % due to higher friction

% Serviceable Available Market (calculated from TAM)
\newcommand{\samHw}{\fpeval{round(\tamHwGlobal * \samHwPercentOfTam / 100, -4)}}

% Target penetration of SAM (%) - based on market benchmarks
% YubiKey: 12% of hardware authentication, Ledger/Trezor: 20-25% of hardware wallets
% Tile: 15% of Bluetooth trackers with retail partnerships
\newcommand{\targetShareHw}{10}  % 10% of SAM with strategic partnership

% Target customer count for hardware (thousands)
\newcommand{\targetHwCountk}{\fpeval{round(\samHw * \targetShareHw / 100 / 1000, 0)}}
