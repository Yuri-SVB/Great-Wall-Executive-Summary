% Document-level Macros
% Semantic markup commands following clean room paradigm
% These commands separate content from formatting

% =============================================================================
% STRUCTURAL ELEMENTS
% =============================================================================

% Executive summary highlighting
\newcommand{\executivehighlight}[1]{%
  \begin{center}
    \colorbox{LightGray}{%
      \begin{minipage}{0.9\textwidth}
        \centering\textbf{\color{PrimaryColor}#1}
      \end{minipage}
    }
  \end{center}
}

% Key takeaway boxes
\newcommand{\keytakeaway}[1]{%
  \begin{quote}
    \color{PrimaryColor}\textbf{Key Takeaway:} 
    \color{TextColor}#1
  \end{quote}
}

% Risk callouts
\newcommand{\riskcallout}[1]{%
  \begin{quote}
    \color{AccentColor}\textbf{Risk Consideration:} 
    \color{TextColor}#1
  \end{quote}
}

% Opportunity highlights
\newcommand{\opportunityhighlight}[1]{%
  \begin{quote}
    \color{SecondaryColor}\textbf{Opportunity:} 
    \color{TextColor}#1
  \end{quote}
}

% =============================================================================
% CONTENT ORGANIZATION
% =============================================================================

% Section introduction paragraph
\newcommand{\sectionintro}[1]{%
  \textit{\color{TextColor}#1}
  \vspace{0.5em}
}

% Subsection summary
\newcommand{\subsectionsummary}[1]{%
  \begin{flushright}
    \textit{\small\color{SecondaryColor}Summary: #1}
  \end{flushright}
}

% Cross-reference with context
\newcommand{\contextref}[2]{%
  \hyperref[#1]{\color{PrimaryColor}#2}%
}

% =============================================================================
% EMPHASIS AND STYLING
% =============================================================================

% Company name styling
\newcommand{\companyname}{%
  \textbf{\color{PrimaryColor}\CompanyName}%
}

% Product/service name styling
\newcommand{\productname}[1]{%
  \textsc{\color{SecondaryColor}#1}%
}

% Technical term definition
\newcommand{\defineterm}[2]{%
  \textbf{#1}: #2%
}

% =============================================================================
% DOCUMENT NAVIGATION
% =============================================================================

% Table of contents with custom formatting
\newcommand{\documenttoc}{%
  \newpage
  \section*{Table of Contents}
  \addcontentsline{toc}{section}{Table of Contents}
  \tableofcontents
  \newpage
}

% List of figures
\newcommand{\documentlof}{%
  \listoffigures
  \addcontentsline{toc}{section}{List of Figures}
}

% List of tables  
\newcommand{\documentlot}{%
  \listoftables
  \addcontentsline{toc}{section}{List of Tables}
}

% =============================================================================
% APPENDIX MANAGEMENT
% =============================================================================

% Appendix section starter
\newcommand{\startappendices}{%
  \appendix
  \renewcommand{\thesection}{\Alph{section}}
  \section*{Appendices}
  \addcontentsline{toc}{section}{Appendices}
}

% Individual appendix
\newcommand{\appendixsection}[2]{%
  \section{#1}\label{app:#2}
}
